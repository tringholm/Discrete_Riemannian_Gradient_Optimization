1% Compute the lowest eigenvalue/vector of a symmetric matrix by minimizing the
  2% Rayleigh quotient over S^{n-1}.
  3
  4function Vhist = eigenValueSphere3(A,phi,tol,dt)
  5%#codegen
  6N = length(phi)+1;
  7Tmax = 2000;
  8residual = inf;
  9tolNewton =  1E-13;
 10
 11dA = diag(A);
 12AU = triu(A,1);
 13Vhist = zeros(Tmax,1);
 14u = euclidCoords(phi);
 15rowsums = u.*(A*u);
 16Vhist(1) = sum(rowsums);
 17% tic
 18sums = zeros(1,5);
 19sums(1) = A(1,1)*u(1)^2;                          % i = k, j = k
 20sums(2) = 0;                                      % i < k, j = k
 21sums(3) = 2*(rowsums(1) - sums(1));               % i = k, j > k
 22sums(4) = 0;                                      % i < k, j > k
 23sums(5) = Vhist(1) - sums(3) - sums(1);           % i > k, j > k
 24trisum = 2*u.*(AU*u);
 25
 26newtoniters = 0;
 27t = 0;
 28for i = 1:Tmax
 29    if  residual < tol
 30        break
 31    end
 32    sum3fact = 1;
 33    ufact = 1;
 34    for j = 1:N-1
 35    
 36        
 37        alpha = 0.1;
 38        dta = dt/alpha;
 39        cp = 1/cos(phi(j)); sp = 1/sin(phi(j));
 40        cpa = cos(phi(j)+alpha); spa = sin(phi(j)+alpha);
 41        cc = cpa*cp; ss = spa*sp; sc = spa*cp; cs = cpa*sp;
 42        
 43        Vd = sums(1)*(cc^2-1) + sums(2)*(cc-1) + sums(3)*(cc*ss - 1) + sums(4)*(ss-1) + sums(5)*(ss^2-1);
 44        V = alpha + dta*Vd;
 45        V_old = inf;
 46        cnt = 0;
 47        while abs(V) > tolNewton && cnt < 10 && V_old - V > 0
 48            dV = sums(1)*(-2*sc*cc) + sums(2)*(-sc) + sums(3)*(cc*cs - sc*ss) + sums(4)*cs + sums(5)*2*cs*ss;
 49            dV = 1+ dta*dV - dta*Vd/alpha;
 50            alpha = alpha - V/dV;
 51            dta = dt/alpha;
 52            cpa = cos(phi(j)+alpha); spa = sin(phi(j)+alpha);
 53            cc = cpa*cp; ss = spa*sp; sc = spa*cp; cs = cpa*sp;
 54            Vd = sums(1)*(cc^2-1) + sums(2)*(cc-1) + sums(3)*(cc*ss-1) + sums(4)*(ss-1) + sums(5)*(ss^2-1);
 55            V_old = V;
 56            V = alpha + dta*Vd;
 57            cnt = cnt + 1;
 58        end
 59        
 60        
 61        newtoniters = newtoniters + cnt;
 62
 63        u(j) = u(j)*cc*ufact;
 64        ufact = ufact*ss;
 65        un = u(j+1)*ufact;
 66        sum3fact = sum3fact*ss^2;
 67        
 68        sums(1) = un*A(j+1,j+1)*un;              % i = k, j = k
 69        sums(2) = 0;
 70
 71        j1 = j+1;
 72        for k = 1:j
 73            sums(2) = sums(2) + A(k,j1)*u(k);
 74        end
 75        sums(2) = 2*un*sums(2);
 76        sums(4) = cc*ss*sums(3) + ss*sums(4) - sums(2);           % i < k, j > k
 77        sums(3) = trisum(j+1)*sum3fact;
 78        sums(5) = ss*ss*sums(5) - sums(3) - sums(1);           % i > k, j > k
 79        phi(j) = phi(j) + alpha;
 80    end
 81    u(end) = u(end)*ufact;
 82    trisum = 2*u.*(AU*u);
 83
 84    dsum = sum(u.*dA.*u);
 85    Vhist(i+1) = sum(trisum)+dsum;
 86    sums(1) = A(1,1)*u(1)^2;                          % i = k, j = k
 87    sums(2) = 0;                                      % i < k, j = k
 88    sums(3) = trisum(1);                              % i = k, j > k
 89    sums(4) = 0;                                      % i < k, j > k
 90    sums(5) = Vhist(i+1) - sums(3) - sums(1);         % i > k, j > k
 91    residual = (Vhist(i) - Vhist(i+1))/abs(Vhist(1));
 92end
 93% toc
 94t
 95disp(['Algorithm DG-MANIFOLD converged with residual ' num2str(residual) ' in ' int2str(i-1) ' steps.'])
 96Vhist = Vhist(~~Vhist);
 97[V,D]= eig(A);
 98min(norm(V(:,1)-u),norm(V(:,1)+u))
 99D(1,1) - Vhist(end)
100newtoniters
101end

103function x = euclidCoords(phi)
104n = length(phi);
105x = zeros(n+1,1);
106x(1) = cos(phi(1));
107sinefact = sin(phi(1));
108for i = 2:n
109    x(i) = cos(phi(i))*sinefact;
110    sinefact = sinefact*sin(phi(i));
111end
112x(end) = sinefact;
113end
